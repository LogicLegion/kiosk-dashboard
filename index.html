import pandas as pd import matplotlib.pyplot as plt import numpy as np from datetime import datetime import plotly.graph_objects as go from plotly.subplots import make_subplots import os # ============================================================ # KIOSK INTELLIGENCE PLATFORM - PYTHON VERSION # FULLY WORKING - No Plotly Errors # ============================================================ # Configuration TRANSACTION_VALUE = 5 HIGH_REJECT_THRESHOLD = 4 SERVICE_OVERDUE_DAYS = 60 def load_and_process_data(csv_path): """Load CSV and calculate all metrics""" print(f"Loading: {csv_path}") df = pd.read_csv(csv_path, encoding='utf-8-sig') # Clean column names df.columns = df.columns.str.strip() def safe_float(val, default=0): try: return float(val) if pd.notna(val) else default except: return default # Extract metrics df['Volume'] = df['1 Day TX'].apply(safe_float) df['Volume_7Day'] = df['7 Day TX'].apply(safe_float) df['Reject_7Day'] = df['7 Day Reject %'].apply(safe_float) df['Uptime_7Day'] = df['7 Day Uptime %'].apply(safe_float) df['Uptime_1Day'] = df['1 Day Uptime %'].apply(safe_float) df['TEMM'] = df['7 Day TEMM'].apply(safe_float) df['BinFull'] = df['Total Bin Full %'].apply(safe_float) df['Latitude'] = df['Latitude'].apply(safe_float) df['Longitude'] = df['Longitude'].apply(safe_float) df['State'] = df['State'].fillna('Unknown') df['Store'] = df['Store'].fillna('Unknown') df['City'] = df['City'].fillna('') df['Address'] = df['Address'].fillna('') # Calculate derived metrics df['Volume_7Day_Avg'] = df['Volume_7Day'] / 7 df['Impact_Score'] = df['Volume'] * (df['Reject_7Day'] / 100) df['Loss_Reject'] = df['Volume'] * (df['Reject_7Day'] / 100) * TRANSACTION_VALUE df['Loss_Downtime'] = df['Volume'] * ((100 - df['Uptime_7Day']) / 100) * 0.8 * TRANSACTION_VALUE df['Daily_Loss'] = df['Loss_Reject'] + df['Loss_Downtime'] df['Monthly_Loss'] = df['Daily_Loss'] * 30 df['Est_Revenue'] = df['Volume'] * TRANSACTION_VALUE df['Net_Revenue'] = df['Est_Revenue'] - df['Daily_Loss'] # Status flags df['Is_High_Reject'] = df['Reject_7Day'] >= HIGH_REJECT_THRESHOLD df['Is_Offline'] = df['Kiosk State'] == 'Offline' # Volume priority volumes = df['Volume'].sort_values(ascending=False).tolist() if volumes: top33_idx = int(len(volumes) * 0.33) - 1 mid33_idx = int(len(volumes) * 0.66) - 1 top33_threshold = volumes[top33_idx] if top33_idx >= 0 else 0 mid33_threshold = volumes[mid33_idx] if mid33_idx >= 0 else 0 else: top33_threshold = mid33_threshold = 0 df['Volume_Priority'] = 'Low' df.loc[df['Volume'] >= top33_threshold, 'Volume_Priority'] = 'High' df.loc[(df['Volume'] < top33_threshold) & (df['Volume'] >= mid33_threshold), 'Volume_Priority'] = 'Mid' # Service days def days_since_service(date_str): if pd.isna(date_str) or date_str == '—': return None try: parts = str(date_str).split('/') if len(parts) == 3: date = datetime(int(parts[2]) + 2000, int(parts[0]), int(parts[1])) return (datetime.now() - date).days except: pass return None df['Days_Since_Service'] = df['Last PM Date'].apply(days_since_service) df['Is_Service_Overdue'] = df['Days_Since_Service'] > SERVICE_OVERDUE_DAYS print(f"Loaded {len(df)} kiosks") print(f" High Reject: {df['Is_High_Reject'].sum()}") print(f" Daily Loss: ${df['Daily_Loss'].sum():,.2f}") return df def create_visualizations(df, output_dir='kiosk_analysis'): """Generate charts for portfolio""" os.makedirs(output_dir, exist_ok=True) print(f"\nGenerating visualizations in '{output_dir}/'...") plt.rcParams['font.family'] = 'sans-serif' plt.rcParams['font.sans-serif'] = ['Arial', 'DejaVu Sans', 'Helvetica'] # 1. Loss by Province province_loss = df.groupby('State')['Daily_Loss'].sum().sort_values(ascending=False).head(10) fig, ax = plt.subplots(figsize=(12, 6)) bars = ax.bar(province_loss.index, province_loss.values, color='#dc2626') ax.set_title('Daily Revenue Loss by Province', fontsize=14, fontweight='bold') ax.set_xlabel('Province') ax.set_ylabel('Daily Loss (CAD)') for bar, val in zip(bars, province_loss.values): ax.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 5, f'${val:.0f}', ha='center', fontsize=9) plt.tight_layout() plt.savefig(f'{output_dir}/loss_by_province.png', dpi=150, bbox_inches='tight') plt.close() print(f" loss_by_province.png") # 2. Top 10 Loss top10 = df.nlargest(10, 'Daily_Loss')[['Store', 'Daily_Loss']] fig, ax = plt.subplots(figsize=(12, 6)) bars = ax.barh(top10['Store'], top10['Daily_Loss'], color='#f97316') ax.set_title('Top 10 Highest Daily Loss', fontsize=14, fontweight='bold') ax.set_xlabel('Daily Loss (CAD)') for bar, val in zip(bars, top10['Daily_Loss']): ax.text(bar.get_width() + 2, bar.get_y() + bar.get_height()/2, f'${val:.0f}', va='center', fontsize=9) plt.tight_layout() plt.savefig(f'{output_dir}/top10_loss.png', dpi=150, bbox_inches='tight') plt.close() print(f" top10_loss.png") # 3. Reject Rate Distribution fig, ax = plt.subplots(figsize=(12, 5)) ax.hist(df['Reject_7Day'], bins=30, color='#3b82f6', edgecolor='white', alpha=0.7) ax.axvline(x=4, color='red', linestyle='--', linewidth=2, label='High Reject Threshold (4%)') ax.set_title('Distribution of 7-Day Reject Rates', fontsize=14, fontweight='bold') ax.set_xlabel('Reject Rate (%)') ax.set_ylabel('Number of Kiosks') ax.legend() plt.tight_layout() plt.savefig(f'{output_dir}/reject_distribution.png', dpi=150, bbox_inches='tight') plt.close() print(f" reject_distribution.png") # 4. High Impact Pie high_impact = (df['Impact_Score'] > 0.8).sum() normal = len(df) - high_impact fig, ax = plt.subplots(figsize=(8, 8)) ax.pie([high_impact, normal], labels=['High Impact', 'Normal'], colors=['#dc2626', '#10b981'], autopct='%1.1f%%', startangle=90) ax.set_title('Kiosks with High Impact Score (>0.8)', fontsize=14, fontweight='bold') plt.tight_layout() plt.savefig(f'{output_dir}/high_impact_pie.png', dpi=150, bbox_inches='tight') plt.close() print(f" high_impact_pie.png") # 5. Uptime vs Loss Scatter fig, ax = plt.subplots(figsize=(12, 6)) scatter = ax.scatter(df['Uptime_7Day'], df['Daily_Loss'], c=df['Volume'], cmap='viridis', alpha=0.6, s=50) ax.set_xlabel('7-Day Uptime (%)') ax.set_ylabel('Daily Loss (CAD)') ax.set_title('Uptime vs Daily Loss', fontsize=14, fontweight='bold') plt.colorbar(scatter, label='Transaction Volume') ax.axvline(x=95, color='red', linestyle='--', alpha=0.5, label='95% Uptime') ax.legend() plt.tight_layout() plt.savefig(f'{output_dir}/uptime_vs_loss.png', dpi=150, bbox_inches='tight') plt.close() print(f" uptime_vs_loss.png") # 6. Service Overdue overdue = df['Is_Service_Overdue'].sum() not_overdue = len(df) - overdue fig, ax = plt.subplots(figsize=(6, 6)) bars = ax.bar(['Service Overdue', 'OK'], [overdue, not_overdue], color=['#dc2626', '#10b981']) ax.set_title('Preventive Maintenance Status', fontsize=14, fontweight='bold') ax.set_ylabel('Number of Kiosks') for bar, val in zip(bars, [overdue, not_overdue]): ax.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 2, str(val), ha='center', fontsize=12, fontweight='bold') plt.tight_layout() plt.savefig(f'{output_dir}/service_overdue.png', dpi=150, bbox_inches='tight') plt.close() print(f" service_overdue.png") print(f"\nAll charts saved to '{output_dir}/'") def create_interactive_dashboard(df, output_file='kiosk_analytics_dashboard.html'): """Create interactive Plotly dashboard - NO ERRORS""" print(f"\nCreating interactive dashboard: {output_file}") # Create subplot figure (no map to avoid Scattergeo errors) fig = make_subplots( rows=2, cols=2, subplot_titles=('Daily Loss by Province', 'Top 10 Daily Loss', 'Reject Rate Distribution', 'Volume Priority Breakdown'), specs=[[{'type': 'bar'}, {'type': 'bar'}], [{'type': 'histogram'}, {'type': 'pie'}]] ) # 1. Loss by Province (top-left) province_loss = df.groupby('State')['Daily_Loss'].sum().sort_values(ascending=False).head(10) fig.add_trace( go.Bar(x=province_loss.index, y=province_loss.values, marker_color='#dc2626', name='Loss'), row=1, col=1 ) # 2. Top 10 Loss (top-right) top10 = df.nlargest(10, 'Daily_Loss') fig.add_trace( go.Bar(x=top10['Daily_Loss'], y=top10['Store'], orientation='h', marker_color='#f97316', name='Loss'), row=1, col=2 ) # 3. Reject Distribution (bottom-left) fig.add_trace( go.Histogram(x=df['Reject_7Day'], nbinsx=30, marker_color='#3b82f6', name='Kiosks'), row=2, col=1 ) # 4. Volume Priority Pie (bottom-right) volume_counts = df['Volume_Priority'].value_counts() colors = {'High': '#dc2626', 'Mid': '#f59e0b', 'Low': '#10b981'} pie_colors = [colors.get(p, '#10b981') for p in volume_counts.index] fig.add_trace( go.Pie(labels=volume_counts.index, values=volume_counts.values, marker_colors=pie_colors), row=2, col=2 ) # Add reference line for reject threshold fig.add_vline(x=4, line_dash="dash", line_color="red", row=2, col=1) # Update layout fig.update_layout( title=dict(text="Kiosk Intelligence Platform", font=dict(size=20)), height=900, showlegend=False ) # Update axes labels fig.update_xaxes(title_text="Province", row=1, col=1) fig.update_yaxes(title_text="Daily Loss (CAD)", row=1, col=1) fig.update_xaxes(title_text="Daily Loss (CAD)", row=1, col=2) fig.update_yaxes(title_text="Store", row=1, col=2) fig.update_xaxes(title_text="Reject Rate (%)", row=2, col=1) fig.update_yaxes(title_text="Number of Kiosks", row=2, col=1) fig.write_html(output_file) print(f"Interactive dashboard saved: {output_file}") return output_file def generate_report(df): """Generate text report""" print("\n" + "="*60) print("KIOSK INTELLIGENCE PLATFORM - DAILY REPORT") print("="*60) print(f"Report Date: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}") print(f"Total Kiosks: {len(df)}") print("-"*60) print("\nFINANCIAL SUMMARY") print(f" Est. Daily Revenue: ${df['Est_Revenue'].sum():,.2f}") print(f" Daily Loss: ${df['Daily_Loss'].sum():,.2f}") print(f" Net Revenue: ${(df['Est_Revenue'].sum() - df['Daily_Loss'].sum()):,.2f}") print(f" Monthly Loss: ${df['Monthly_Loss'].sum():,.2f}") print("\nOPERATIONAL ALERTS") print(f" High Reject (>=4%): {df['Is_High_Reject'].sum()}") print(f" Offline: {(df['Kiosk State'] == 'Offline').sum()}") print(f" High Volume (Top 33%): {(df['Volume_Priority'] == 'High').sum()}") print(f" Service Overdue: {df['Is_Service_Overdue'].sum()}") print("\nTOP 5 LOSS") top5 = df.nlargest(5, 'Daily_Loss')[['Store', 'City', 'State', 'Daily_Loss']] for i, (_, row) in enumerate(top5.iterrows(), 1): print(f" {i}. {row['Store']} - {row['City']}, {row['State']}: ${row['Daily_Loss']:.0f}/day") print("\n" + "="*60) # ============================================================ # MAIN EXECUTION # ============================================================ if __name__ == "__main__": import sys csv_file = "Canada Kiosk Status.csv" if len(sys.argv) > 1: csv_file = sys.argv[1] try: df = load_and_process_data(csv_file) generate_report(df) create_visualizations(df) create_interactive_dashboard(df, "kiosk_analytics_dashboard.html") output_excel = f"kiosk_export_{datetime.now().strftime('%Y%m%d_%H%M%S')}.xlsx" df.to_excel(output_excel, index=False) print(f"\nExcel export: {output_excel}") print("\n" + "="*50) print("SUCCESS!") print(f" Dashboard: kiosk_analytics_dashboard.html") print(f" Charts: ./kiosk_analysis/ (6 images)") print(f" Excel: {output_excel}") print("="*50) except FileNotFoundError: print(f"Error: Could not find '{csv_file}'") print(f" Usage: python team_priority.py 'your_file.csv'") except Exception as e: print(f"Error: {e}")
